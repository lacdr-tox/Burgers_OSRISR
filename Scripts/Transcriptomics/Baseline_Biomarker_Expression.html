countdata_path <- "E:/TD22 HLCs TempoSeq/DATA/upload_countdata_cpm_fprobe_fsample_sumprobe.txt"
metadata_path <- "E:/TD22 HLCs TempoSeq/DATA/upload_metadata.txt"
outputDIR <- "E:/TD22 HLCs TempoSeq/DATA_TD53-57"
countdata_filtered <- read.table(countdata_path, header = TRUE, sep = "\t", stringsAsFactors = FALSE)
metadata_filtered <- read.table(metadata_path, header = TRUE, sep = "\t", stringsAsFactors = FALSE)

##Urea cycle related genes: baseline expression across liver models

countdata_filtered_urea <- countdata_filtered %>%
  filter(gene_symbol %in% c("CPS1", "ASS1", "ASL","ARG1"))  

# Log2 normalization of the CPM counts (adding 1 to avoid log(0))
countdata_filtered_log2 <- countdata_filtered_urea %>%
  mutate_at(vars(-gene_symbol), funs(log2(. + 1)))
## Warning: `funs()` was deprecated in dplyr 0.8.0.
## ℹ Please use a list of either functions or lambdas:
## 
## # Simple named list: list(mean = mean, median = median)
## 
## # Auto named with `tibble::lst()`: tibble::lst(mean, median)
## 
## # Using lambdas list(~ mean(., trim = .2), ~ median(., na.rm = TRUE))
## Call `lifecycle::last_lifecycle_warnings()` to see where this warning was
## generated.
# Reshape the countdata to long format
countdata_long <- countdata_filtered_log2 %>%
  gather(key = "sample_name", value = "log2_count", -gene_symbol)

# Add cell type information to the data (from metadata)
countdata_long <- countdata_long %>%
  left_join(metadata_filtered %>%
              select(sample_name, cell_ID), by = "sample_name")


# Calculate summary statistics for error bars (mean ± SE)
summary_data <- countdata_long %>%
  group_by(gene_symbol, cell_ID) %>%
  summarise(
    mean = mean(log2_count),
    se = sd(log2_count) / sqrt(n()),
    .groups = "drop"
  )

p1=ggplot() +
  geom_bar(
    data = summary_data,
    aes(x = gene_symbol, y = mean, fill = cell_ID),
    stat = "identity",
    position = position_dodge(0.7),
    width = 0.6,
    color = "black"
  ) +
  geom_errorbar(
    data = summary_data,
    aes(x = gene_symbol, ymin = mean - se, ymax = mean + se, group = cell_ID),
    position = position_dodge(0.7),
    width = 0.2,
    color = "black"
  ) +
  geom_jitter(
    data = countdata_long,
    aes(x = gene_symbol, y = log2_count, fill = cell_ID),
    position = position_jitterdodge(jitter.width = 0.1, dodge.width = 0.7),
    shape = 21,
    size = 1.5,
    stroke = 0.2,
    alpha = 0.7,
    color = "black"
  ) +
  scale_fill_viridis_d(option = "D", name = "Cell Type") +
  scale_x_discrete(limits = c("CPS1", "ASS1", "ASL", "ARG1")) +
  coord_cartesian(ylim = c(0, 8)) +
  labs(
    x = "Gene",
    y = expression(Log[2]~"Normalized Counts"),
    title = "Baseline expression of urea cycle-related genes"
  ) +
  theme_classic(base_size = 12) +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1, vjust = 1),
    legend.position = "top",
    plot.title = element_text(face = "bold", hjust = 0.5, size = 14)
  )


print(p1)
